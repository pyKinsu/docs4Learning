---
title: "Common Includes"
description: "Frequently used C header files"
---

```c
#include <stdio.h>    // printf, scanf, fgets
#include <stdlib.h>   // malloc, free, exit, atoi
#include <string.h>   // strlen, strcpy, strcmp, memset
#include <stdint.h>   // int32_t, uint8_t, etc.
#include <stdbool.h>  // bool, true, false (C99+)
#include <math.h>     // sqrt, pow, fabs  (link with -lm)
#include <ctype.h>    // isdigit, isalpha, toupper, tolower
#include <limits.h>   // INT_MAX, INT_MIN, CHAR_MAX
```
